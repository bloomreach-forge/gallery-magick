37  /**
38   * Utility to identify or resize images using Scalr library.
39   */
40  public class ScalrProcessorUtils {

42      private ScalrProcessorUtils() {
43      }

45      /**
46       * Detects the {@code sourceFile} and returns the size dimension from it.
47       * @param sourceFile source image file
48       * @return Detects the {@code sourceFile} and returns the size dimension from it
49       * @throws IOException if IO exception occurs
50       */
51      public static ImageDimension identifyDimension(File sourceFile) throws IOException {
52          ImageDimension dimension = null;
53          String extension = FilenameUtils.getExtension(sourceFile.getName());
54          Iterator<ImageReader> it = ImageIO.getImageReadersBySuffix(extension);
55  
56          if (!it.hasNext()) {
57              throw new IllegalArgumentException("Unsupported file name extension: " + sourceFile);
58          }
59  
60          ImageReader reader = null;
61  
62          try {
63              reader = getImageReader(sourceFile);
64              int width = reader.getWidth(reader.getMinIndex());
65              int height = reader.getHeight(reader.getMinIndex());
66              dimension = ImageDimension.from(width, height);
67          } finally {
68              if (reader != null) {
69                  reader.dispose();
70              }
71          }
72  
73          return dimension;
74      }

76      /**
77       * Resize the given image {@code sourceFile} with resizing it to {@code width} and {@code height}
78       * and store the resized image to {@code targetFile}.
79       * @param sourceFile source image file
80       * @param targetFile target image file
81       * @param dimension image dimension
82       * @throws IOException if IO exception occurs
83       */
84      public static void resizeImage(File sourceFile, File targetFile, ImageDimension dimension) throws IOException {
85          resizeImage(sourceFile, targetFile, dimension, (String[]) null);
86      }

88      /**
89       * Resize the given image {@code sourceFile} with resizing it to {@code width} and {@code height}
90       * and store the resized image to {@code targetFile}, with appending {@code extraOptions} in the command line if provided.
91       * @param sourceFile source image file
92       * @param targetFile target image file
93       * @param dimension image dimension
94       * @param extraOptions extra command line options
95       * @throws IOException if IO exception occurs
96       */
97      public static void resizeImage(File sourceFile, File targetFile, ImageDimension dimension, String... extraOptions)
98              throws IOException {
99          if (dimension == null) {
100             throw new IllegalArgumentException("Invalid dimension: " + dimension);
101         }
102 
103         ImageReader reader = null;
104         ImageWriter writer = null;
105 
106         try {
107             reader = getImageReader(sourceFile);
108 
109             if (reader == null) {
110                 throw new IllegalArgumentException("Unsupported image file name extension for reading: " + sourceFile);
111             }
112 
113             writer = getImageWriter(targetFile);
114 
115             if (writer == null) {
116                 throw new IllegalArgumentException("Unsupported image file name extension for writing: " + targetFile);
117             }
118 
119             BufferedImage sourceImage = reader.read(0);
120             BufferedImage resizedImage = Scalr.resize(sourceImage, Scalr.Method.QUALITY, Scalr.Mode.AUTOMATIC,
121                     dimension.getWidth(), dimension.getHeight());
122 
123             final ImageWriteParam writeParam = writer.getDefaultWriteParam();
124 
125             if (writeParam.canWriteCompressed()) {
126                 String[] compressionTypes = writeParam.getCompressionTypes();
127 
128                 if (compressionTypes != null && compressionTypes.length > 0) {
129                     writeParam.setCompressionMode(ImageWriteParam.MODE_EXPLICIT);
130                     writeParam.setCompressionType(compressionTypes[0]);
131                     writeParam.setCompressionQuality(1.0f);
132                 }
133             }
134 
135             final IIOImage iioImage = new IIOImage(resizedImage, null, null);
136             writer.write(null, iioImage, writeParam);
137         } finally {
138             if (reader != null) {
139                 reader.dispose();
140             }
141             if (writer != null) {
142                 writer.dispose();
143             }
144         }
145     }

147     /**
148      * Creates and returns an {@link ImageReader} instance from the {@code sourceFile}.
149      * @param sourceFile source file
150      * @return an {@link ImageReader} instance
151      * @throws IOException if IOException occurs
152      */
153     private static ImageReader getImageReader(File sourceFile) throws IOException {
154         ImageReader reader = null;
155         String extension = FilenameUtils.getExtension(sourceFile.getName());
156         Iterator<ImageReader> it = ImageIO.getImageReadersBySuffix(extension);
157 
158         if (it.hasNext()) {
159             reader = it.next();
160             ImageInputStream input = new FileImageInputStream(sourceFile);
161             reader.setInput(input);
162         }
163 
164         return reader;
165     }
166 
167     /**
168      * Creates and returns an {@link ImageWriter} instance from the {@code targetFile}.
169      * @param targetFile target file
170      * @return an {@link ImageWriter} instance
171      * @throws IOException if IOException occurs
172      */
173     private static ImageWriter getImageWriter(File targetFile) throws IOException {
174         ImageWriter writer = null;
175         String extension = FilenameUtils.getExtension(targetFile.getName());
176         Iterator<ImageWriter> it = ImageIO.getImageWritersBySuffix(extension);
177 
178         if (it.hasNext()) {
179             writer = it.next();
180             ImageOutputStream output = new FileImageOutputStream(targetFile);
181             writer.setOutput(output);
182         }
183 
184         return writer;
185     }
